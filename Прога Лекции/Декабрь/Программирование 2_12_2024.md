# Строки
Ввод с консоли
```cpp
#include <iostream>
#include <string.h>

int main() {
	std::string str;
	getline(std::cin, str);
	std::cout << str;
	return 0;
}
```
Ошибка, считается перевод на следующую строку
```cpp
#include <iostream>
#include <string.h>

int main() {
	int n;
	std::cin >> n;
	std::string str;
	getline(std::cin, str);
	std::cout << str;
	return 0;
}
```
Нормальное считывание
```cpp
#include <iostream>
#include <string.h>

int main() {
	int n;
	std::cin >> n;
	std::cin.get(); // очистка буфера
	std::string str;
	getline(std::cin, str);
	std::cout << str;
	return 0;
}
```
Три параметра getline
```cpp
#include <iostream>
#include <string.h>

int main() {
	int n;
	std::cin >> n;
	std::cin.get();
	std::string str;
	getline(std::cin, str, '#'); // Считывает, пока не будет введён символ #
	std::cout << str;
	return 0;
}
```

# Методы строк
```cpp
#include <iostream>
#include <string.h>

int main() {
	std::string str = "Hello, world!";
	std::cout << str.length;
	std::cout << *str.end(); // будет ошибка, попытка вывести элемент, которого нет в строке
	std::cout << *str.begin(); // выведет первый элемент
	std::cout << str[14]; // выход за пределы типа
	std::cout << str.at(14); // вышло исключение index_out_of_range
	return 0;
}
```
## Вставка
```cpp

	std::string str = "Hello, world!";

}
```
